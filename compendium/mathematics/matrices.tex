\section*{Matrices}

\subsection*{Basic matrix}

\[
\begin{matrix}
	a_{11} & \cdots & a_{1k}\\
	\vdots & \ddots & \vdots\\
	a_{k1} & \cdots & a_{kk}
\end{matrix}
\]

\subsection*{Matrix with braces}

\[
\begin{pmatrix}
	a_{11} & \cdots & a_{1k}\\
	\vdots & \ddots & \vdots\\
	a_{k1} & \cdots & a_{kk}
\end{pmatrix}
\]

A lot of environments exist with different brackets.

\[
\begin{vmatrix}
	a_{11} & \cdots & a_{1k}\\
	\vdots & \ddots & \vdots\\
	a_{k1} & \cdots & a_{kk}
\end{vmatrix}
\]

\[
\begin{Vmatrix}
	a_{11} & \cdots & a_{1k}\\
	\vdots & \ddots & \vdots\\
	a_{k1} & \cdots & a_{kk}
\end{Vmatrix}
\]

\[
\begin{bmatrix}
	a_{11} & \cdots & a_{1k}\\
	\vdots & \ddots & \vdots\\
	a_{k1} & \cdots & a_{kk}
\end{bmatrix}
\]

\[
\begin{Bmatrix}
	a_{11} & \cdots & a_{1k}\\
	\vdots & \ddots & \vdots\\
	a_{k1} & \cdots & a_{kk}
\end{Bmatrix}
\]
