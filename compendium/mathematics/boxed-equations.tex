\section*{Boxed equation}

\begin{alignat*}{2}
	&& \sqrt{4 x^2} - 1 &= 0 \\
	&\Rightarrow & \sqrt{4} \sqrt{x^2} &= 1 \\
	&\Rightarrow & 2 \left|x\right| &= 1 \\
	&\Rightarrow & x &= \pm \frac{1}{2}
\end{alignat*}

The final result is

\begin{equation}
	\boxed{x = \pm \frac{1}{2}}
\end{equation}
