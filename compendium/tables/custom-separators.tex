	% The difference is !{} has a the advantage compared to @{} to insert a space before and after.
	\begin{tabular}{| l | r @{,} l !{\$} |}
		\hline
		MacBook & 1299 & 00 \\
		\hline
		MacBook Air 11-inch & 899 & 99 \\
		\hline
		MacBook Air 13-inch & 999 & 99 \\
		\hline
	\end{tabular}
	\caption{Table using custom separators}
\end{table}
